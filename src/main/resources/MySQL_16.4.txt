SELECT IL.NAME, I.SUMMARY
FROM ISSUES I
         JOIN ISSUESLISTS IL ON I.ISSUESLIST_ID = IL.ID;

SELECT U.FIRSTNAME, U.LASTNAME, I.SUMMARY
FROM ISSUES I
         JOIN USERS U on I.USER_ID_ASSIGNEDTO = U.ID
ORDER BY U.LASTNAME, U.FIRSTNAME;

SELECT U.FIRSTNAME, U.LASTNAME, COUNT(*) AS ISSUES_NUMBER
FROM ISSUES I
         JOIN USERS U on I.USER_ID_ASSIGNEDTO = U.ID
GROUP BY USER_ID_ASSIGNEDTO
ORDER BY ISSUES_NUMBER;
